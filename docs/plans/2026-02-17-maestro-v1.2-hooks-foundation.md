# Maestro v1.2 — Hooks Foundation Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add hooks-based lifecycle middleware to Maestro for tool permission enforcement, dynamic context injection, output validation, and thinking level control.

**Architecture:** Five hook handlers (SessionStart, BeforeToolSelection, BeforeTool, BeforeAgent, AfterAgent, BeforeModel) backed by a compiled permissions manifest. Parallel dispatch enhanced with per-agent environment variables. All 12 agents updated with `model: auto` and canonical tool names.

**Tech Stack:** Shell scripts (bash), JSON, YAML frontmatter, TOML. No new runtime dependencies.

**Design Reference:** `docs/plans/2026-02-17-maestro-v2-hybrid-layered-architecture-design.md`

**Critical API Constraints Discovered:**
- Agent frontmatter uses `.strict()` Zod validation — NO extra fields allowed (no `thinking_level` in frontmatter)
- `BeforeTool` hook input does NOT include `agentName` — must track via env var (parallel) or state file (sequential)
- `BeforeToolSelection` hook can restrict `allowedFunctionNames` BEFORE the model considers tools — preferred over blocking
- Sequential `delegate_to_agent` already enforces tool isolation via isolated `ToolRegistry` — hooks are defense-in-depth for parallel mode
- `search_file_content` is a legacy alias for `grep_search` — update to canonical name
- Hooks communicate via JSON on stdin/stdout. Exit code 0 = success, 2+ = blocking error

```bash
#!/usr/bin/env bash
set -euo pipefail

MAESTRO_HOOKS_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
MAESTRO_ROOT="$(cd "$MAESTRO_HOOKS_DIR/.." && pwd)"

read_stdin() {
  local input=""
  if [ -t 0 ]; then
    echo "{}"
    return
  fi
  input=$(cat)
  echo "$input"
}

json_get() {
  local json="$1"
  local key="$2"
  echo "$json" | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('$key',''))" 2>/dev/null || echo ""
}

json_get_nested() {
  local json="$1"
  shift
  local keys="$*"
  echo "$json" | python3 -c "
import sys, json
d = json.load(sys.stdin)
keys = '$keys'.split()
for k in keys:
    if isinstance(d, dict):
        d = d.get(k, '')
    else:
        d = ''
        break
print(d if isinstance(d, str) else json.dumps(d))
" 2>/dev/null || echo ""
}

respond_allow() {
  local msg="${1:-}"
  if [ -n "$msg" ]; then
    echo "{\"decision\":\"allow\",\"systemMessage\":\"$msg\"}"
  else
    echo "{\"decision\":\"allow\"}"
  fi
}

respond_block() {
  local reason="$1"
  echo "{\"decision\":\"block\",\"reason\":\"$reason\"}"
}

respond_with_context() {
  local event="$1"
  local context="$2"
  echo "{\"hookSpecificOutput\":{\"hookEventName\":\"$event\",\"additionalContext\":\"$context\"}}"
}

log_hook() {
  local level="$1"
  local msg="$2"
  echo "[$level] maestro: $msg" >&2
}

get_active_agent() {
  local session_id="$1"
  local state_dir="/tmp/maestro-hooks"
  local agent_file="$state_dir/$session_id/active-agent"
  if [ -f "$agent_file" ]; then
    cat "$agent_file"
  else
    echo ""
  fi
}

set_active_agent() {
  local session_id="$1"
  local agent_name="$2"
  local state_dir="/tmp/maestro-hooks"
  mkdir -p "$state_dir/$session_id"
  echo "$agent_name" > "$state_dir/$session_id/active-agent"
}

clear_active_agent() {
  local session_id="$1"
  local state_dir="/tmp/maestro-hooks"
  rm -f "$state_dir/$session_id/active-agent"
}

load_permissions() {
  local perms_file="$MAESTRO_HOOKS_DIR/permissions.json"
  if [ -f "$perms_file" ]; then
    cat "$perms_file"
  else
    echo "{}"
  fi
}

get_agent_tools() {
  local permissions="$1"
  local agent_name="$2"
  echo "$permissions" | python3 -c "
import sys, json
perms = json.load(sys.stdin)
tools = perms.get('$agent_name', [])
print(' '.join(tools))
" 2>/dev/null || echo ""
}

is_tool_allowed() {
  local permissions="$1"
  local agent_name="$2"
  local tool_name="$3"
  echo "$permissions" | python3 -c "
import sys, json
perms = json.load(sys.stdin)
agent_tools = perms.get('$agent_name', None)
if agent_tools is None:
    print('unknown_agent')
elif '$tool_name' in agent_tools:
    print('allowed')
elif '$tool_name'.startswith('maestro__'):
    mcp_base = '$tool_name'.replace('maestro__', '')
    if mcp_base in agent_tools or '$tool_name' in agent_tools:
        print('allowed')
    else:
        print('blocked')
else:
    print('blocked')
" 2>/dev/null || echo "allowed"
}
```

```bash
#!/usr/bin/env bash
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
AGENTS_DIR="$(cd "$SCRIPT_DIR/../agents" && pwd)"
OUTPUT="${MAESTRO_PERMISSIONS_OUTPUT:-$SCRIPT_DIR/permissions.json}"

if [ ! -d "$AGENTS_DIR" ]; then
  echo "Error: agents directory not found at $AGENTS_DIR" >&2
  exit 1
fi

python3 -c "
import os, re, json, sys, yaml

agents_dir = '$AGENTS_DIR'
output_path = '$OUTPUT'
permissions = {}

frontmatter_re = re.compile(r'^---\r?\n(.*?)\r?\n---', re.DOTALL)

for filename in sorted(os.listdir(agents_dir)):
    if not filename.endswith('.md'):
        continue

    filepath = os.path.join(agents_dir, filename)
    with open(filepath, 'r') as f:
        content = f.read()

    match = frontmatter_re.match(content)
    if not match:
        print(f'Warning: No frontmatter in {filename}', file=sys.stderr)
        continue

    try:
        meta = yaml.safe_load(match.group(1))
    except yaml.YAMLError as e:
        print(f'Warning: Invalid YAML in {filename}: {e}', file=sys.stderr)
        continue

    name = meta.get('name', '')
    tools = meta.get('tools', [])

    if not name:
        print(f'Warning: No name in {filename}', file=sys.stderr)
        continue

    if not tools:
        print(f'Warning: No tools specified for {name} — will have unrestricted access', file=sys.stderr)

    permissions[name] = tools

os.makedirs(os.path.dirname(output_path) or '.', exist_ok=True)
with open(output_path, 'w') as f:
    json.dump(permissions, f, indent=2, sort_keys=True)

print(f'Generated permissions for {len(permissions)} agents -> {output_path}', file=sys.stderr)
"
```

```bash
#!/usr/bin/env bash
set -euo pipefail

HOOK_DIR="$(cd "$(dirname "$0")" && pwd)"
source "$HOOK_DIR/lib/common.sh"

INPUT=$(read_stdin)
SESSION_ID=$(json_get "$INPUT" "session_id")
TOOL_NAME=$(json_get "$INPUT" "tool_name")

AGENT_NAME="${MAESTRO_CURRENT_AGENT:-}"

if [ -z "$AGENT_NAME" ]; then
  AGENT_NAME=$(get_active_agent "$SESSION_ID")
fi

if [ -z "$AGENT_NAME" ]; then
  respond_allow
  exit 0
fi

PERMISSIONS=$(load_permissions)
RESULT=$(is_tool_allowed "$PERMISSIONS" "$AGENT_NAME" "$TOOL_NAME")

case "$RESULT" in
  "allowed")
    respond_allow
    ;;
  "unknown_agent")
    log_hook "WARN" "Unknown agent '$AGENT_NAME' — allowing tool '$TOOL_NAME'"
    respond_allow
    ;;
  "blocked")
    log_hook "WARN" "BLOCKED: Agent '$AGENT_NAME' attempted to use '$TOOL_NAME'"
    respond_block "$AGENT_NAME is not authorized to use $TOOL_NAME. Allowed tools: $(get_agent_tools "$PERMISSIONS" "$AGENT_NAME")"
    ;;
  *)
    respond_allow
    ;;
esac
```

```bash
set -euo pipefail

HOOK_DIR="$(cd "$(dirname "$0")" && pwd)"
source "$HOOK_DIR/lib/common.sh"

INPUT=$(read_stdin)
SESSION_ID=$(json_get "$INPUT" "session_id")
PROMPT=$(json_get "$INPUT" "prompt")

if [ -n "$AGENT_NAME" ]; then
  set_active_agent "$SESSION_ID" "$AGENT_NAME"
  log_hook "INFO" "BeforeAgent: Tracking agent '$AGENT_NAME' [session=$SESSION_ID]"
fi

CONTEXT_PARTS=""

# Inject active session state if it exists
```

```bash
run_with_timeout() {
  local timeout_secs="$1"
  shift

  if command -v timeout &>/dev/null; then
    timeout "$timeout_secs" "$@"
    return $?
  fi

  "$@" &
  local pid=$!

  (
    sleep "$timeout_secs"
    if kill -0 "$pid" 2>/dev/null; then
      log "WARN" "Process $pid timed out after ${timeout_secs}s, sending SIGTERM"
      kill -TERM "$pid" 2>/dev/null
      sleep 5
      if kill -0 "$pid" 2>/dev/null; then
        log "WARN" "Process $pid still running, sending SIGKILL"
        kill -KILL "$pid" 2>/dev/null
      fi
    fi
  ) &
  local watchdog_pid=$!

  wait "$pid" 2>/dev/null
  local exit_code=$?

  kill "$watchdog_pid" 2>/dev/null
  wait "$watchdog_pid" 2>/dev/null

  return "$exit_code"
}
```
